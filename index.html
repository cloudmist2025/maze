  • ${item.name} ${item.score.toFixed(2)}s
    • ` ).join(''); } else { list.innerHTML = "
  • 暂无排名数据
    • "; } } catch (e) { console.log("获取排名失败，请检查数据库 Policy 设置"); } } // --- 渲染与控制 (保持不变) --- function draw() { ctx.fillStyle = "#111"; ctx.fillRect(0, 0, canvas.width, canvas.height); for (let y = 0; y < ROWS; y++) { for (let x = 0; x < COLS; x++) { if (maze[y][x] === 0) { ctx.fillStyle = "#222"; ctx.fillRect(x * TILE_SIZE, y * TILE_SIZE, TILE_SIZE, TILE_SIZE); } } } const px = player.x * TILE_SIZE + TILE_SIZE/2; const py = player.y * TILE_SIZE + TILE_SIZE/2; ctx.save(); ctx.beginPath(); ctx.rect(0, 0, canvas.width, canvas.height); ctx.arc(px, py, VISION, 0, Math.PI * 2, true); ctx.fillStyle = "black"; ctx.fill(); ctx.restore(); ctx.fillStyle = "#00bfff"; ctx.beginPath(); ctx.arc(px, py, TILE_SIZE/3, 0, Math.PI*2); ctx.fill(); ctx.fillStyle = "#2ecc71"; ctx.fillRect(startPos.x*TILE_SIZE+4, startPos.y*TILE_SIZE+4, TILE_SIZE-8, TILE_SIZE-8); ctx.fillStyle = "#ff4757"; ctx.fillRect(exitPos.x*TILE_SIZE+4, exitPos.y*TILE_SIZE+4, TILE_SIZE-8, TILE_SIZE-8); } window.addEventListener('keydown', (e) => { if(isFinished) return; if(["ArrowUp","ArrowDown","ArrowLeft","ArrowRight"].includes(e.key)) { e.preventDefault(); if(!startTime) { startTime = Date.now(); timerInterval = setInterval(() => { document.getElementById('timerDisplay').innerText = ((Date.now()-startTime)/1000).toFixed(2); }, 50); } } let nx = player.x, ny = player.y; if (e.key === 'ArrowUp') ny--; else if (e.key === 'ArrowDown') ny++; else if (e.key === 'ArrowLeft') nx--; else if (e.key === 'ArrowRight') nx++; if (maze[ny] && maze[ny][nx] === 0) { player.x = nx; player.y = ny; draw(); if (player.x === exitPos.x && player.y === exitPos.y) win(); } }); async function win() { isFinished = true; clearInterval(timerInterval); const finalTime = document.getElementById('timerDisplay').innerText; document.getElementById('resultText').innerText = `你的成绩：${finalTime} 秒`; document.getElementById('winOverlay').style.display = 'flex'; const playerName = prompt("恭喜通关！请输入你的大名上传成绩：") || "神秘玩家"; await saveScoreToCloud(playerName, finalTime); } // 确保初始加载 initMaze(); draw();
